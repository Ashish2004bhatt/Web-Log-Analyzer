    // 🌙 Dark Mode Toggle
    function toggleDarkMode() {
        const isDarkMode = document.body.classList.toggle("dark-mode");
        const color = isDarkMode ? 'rgba(255, 255, 255, 0.8)' : 'rgba(0, 0, 0, 0.8)';
        updateChartColors(color);

    // Function to handle the animated counting effect for stats
    function animateCount(elementId, targetValue, sparklineElementId) {
        const element = document.getElementById(elementId);
        const sparklineEl = document.getElementById(sparklineElementId);
        let start = 0;
        const duration = 1500;
        const step = Math.ceil(targetValue / (duration / 10));

        // Anomaly Detection Simulation: Flag Success Count if target is unusually high (e.g., > 150k)
        if (elementId === 'successCount' && targetValue > 150000) {
            document.getElementById('statSuccess').classList.add('anomaly-flash');
            sparklineEl.textContent = 'SPIKE! ⚠️';
            sparklineEl.classList.remove('bg-gray-100');
            sparklineEl.classList.add('bg-red-500', 'text-white');
        } else if (elementId === 'successCount') {
            document.getElementById('statSuccess').classList.remove('anomaly-flash');
            sparklineEl.textContent = '+3.5% ▲';
            sparklineEl.classList.add('bg-gray-100');
            sparklineEl.classList.remove('bg-red-500', 'text-white');
        }

        const timer = setInterval(() => {
            start += step;
            if (start > targetValue) {
                start = targetValue;
                clearInterval(timer);
            }
            element.textContent = start.toLocaleString();
        }, 10);
    }

    // Hide the loading spinner after initial data is loaded
    function hideLoadingSpinner() {
        const spinner = document.getElementById('loading-spinner');
        spinner.style.opacity = '0';

            document.body.classList.add('fade-in');
        }, 300);
    }

    // --- Chart.js Functions ---

    // Function to update chart configuration for dark mode
    function updateChartColors(color) {
        [barChartInstance, pieChartInstance, latencyChartInstance].forEach(chart => {
            if (chart) {
                chart.options.scales.x.ticks.color = color;
                chart.options.scales.y.ticks.color = color;
                chart.options.plugins.legend.labels.color = color;
                chart.update();
            }
        });
    }

    // Function to animate the threat level (Also used for AI Anomaly Detection simulation for the gauge)
    function animateThreat() {
        const threatValueEl = document.getElementById('threatValue');
        let newThreatLevel = Math.floor(Math.random() * 100);

        // Update Gauge
        gaugeInstance.data.datasets[0].data = [newThreatLevel, 100 - newThreatLevel];
        
        // Color changes based on severity
        let color = '#1abc9c'; // Low
        if (newThreatLevel > 40) color = '#f39c12'; // Medium
        if (newThreatLevel > 75) color = '#e74c3c'; // High (Anomaly detected)
        
        gaugeInstance.data.datasets[0].backgroundColor = [color, 'rgba(221, 221, 221, 0.3)'];
        gaugeInstance.update();

        // Update Text and Color
        threatValueEl.textContent = newThreatLevel;
        threatValueEl.className = `text-[${color}] transition-colors duration-500`;

        // Simulate anomaly detection spike in the success counter occasionally
        const currentSuccessTarget = parseInt(document.getElementById('successCount').getAttribute('data-target'));
        const newSuccessTarget = newThreatLevel > 85 ? 180 : 120; // Spike the target if threat is high
        document.getElementById('successCount').setAttribute('data-target', newSuccessTarget);
        
        // Restart count animation to reflect change (and anomaly flash if needed)
        animateCount('successCount', newSuccessTarget * 1000, 'sparkSuccess'); 
    }

    // 🗺️ Initialize Leaflet Map
    function initMap() {
        mapInstance = L.map('map').setView([20.5937, 78.9629], 3);
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap contributors' }).addTo(mapInstance);

        const requestData = [
            { lat: 28.6139, lng: 77.2090, count: 150, city: "New Delhi" },
            { lat: 37.7749, lng: -122.4194, count: 230, city: "San Francisco" },
            { lat: 51.5074, lng: -0.1278, count: 180, city: "London" },
            { lat: 40.7128, lng: -74.0060, count: 320, city: "New York" },
            { lat: 34.0522, lng: -118.2437, count: 90, city: "Los Angeles" }
        ];

        requestData.forEach(item => {
            L.circleMarker([item.lat, item.lng], { radius: 5 + item.count / 50, color: '#0078ff', fillColor: '#0078ff', fillOpacity: 0.7 })
              .addTo(mapInstance).bindPopup(`<strong>${item.city}</strong>: ${item.count} Requests`);
        });

        setTimeout(() => mapInstance.invalidateSize(), 500);
    }
    
    // 🔥 Initialize Heatmap (Time-of-Day)
    function initHeatmap() {
        const heatmapGrid = document.getElementById('heatmapGrid');
        heatmapGrid.innerHTML = '';
        const days = ['Day', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'];
        
        // Simulate higher traffic during peak hours (10-18) and weekdays
        const data = Array(7).fill(0).map(() => Array(24).fill(0).map((_, hour) => {
            if (hour >= 10 && hour <= 18) return Math.floor(Math.random() * 80) + 20; // Peak traffic
            return Math.floor(Math.random() * 40) + 10; // Low traffic
        }));

        // Render Headers (Hours 00-23)
        heatmapGrid.innerHTML += '<div class="heatmap-header bg-transparent shadow-none border-0"></div>'; // Empty top-left cell
        for (let h = 0; h < 24; h++) {
            heatmapGrid.innerHTML += `<div class="heatmap-header">${h}</div>`;
        }

        // Render Day rows and data cells
        for (let d = 1; d <= 7; d++) {
            heatmapGrid.innerHTML += `<div class="heatmap-header">${days[d]}</div>`;
            for (let h = 0; h < 24; h++) {
                const count = data[d - 1][h];
                const intensity = Math.min(255, Math.floor(count * 3)); // Max intensity at ~85
                const green = 255 - intensity;
                const blue = 255 - intensity;
                const cellColor = `rgb(255, ${green}, ${blue})`; // Scale from White/Gray to Red
                
                const textColor = count > 80 ? 'text-white' : 'text-gray-800 dark:text-gray-100';

                heatmapGrid.innerHTML += `<div class="heatmap-cell shadow-sm ${textColor}" style="background-color: ${cellColor};" title="${days[d]} ${h}:00: ${count} reqs"></div>`;
            }

    // 📄 Export to PDF Function
    async function exportToPDF() {
        const messageEl = document.getElementById('pdf-message');
        messageEl.style.opacity = '1';
        messageEl.style.transform = 'translateX(0)';
        
        // Select the main content area to capture (excluding header/nav)
        const element = document.body; 
        
        // Fix for leaflet map rendering in canvas (need to hide controls temporarily)
        const controls = document.querySelectorAll('.leaflet-control-container');
        controls.forEach(c => c.style.visibility = 'hidden');

        try {
            const canvas = await html2canvas(element, { 
                scale: 2, 
                useCORS: true,
                windowWidth: document.documentElement.offsetWidth,
                windowHeight: document.documentElement.offsetHeight,
                scrollX: 0,
                scrollY: 0
            });
            const imgData = canvas.toDataURL('image/png');
            const pdf = new jsPDF('p', 'mm', 'a4');
            const imgWidth = 210; 
            const pageHeight = 295; 
            const imgHeight = canvas.height * imgWidth / canvas.width;
            let heightLeft = imgHeight;
            let position = 0;

            // Add first page
            pdf.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);
            heightLeft -= pageHeight;

            // Handle multi-page content
            while (heightLeft >= 0) {
                position = heightLeft - imgHeight;
                pdf.addPage();
                pdf.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);
                heightLeft -= pageHeight;
            }

            pdf.save('Log_Analytics_Dashboard_Report.pdf');

        } catch (error) {
            console.error("PDF Export failed:", error);
            // Show a simple failure message instead of alert
            messageEl.textContent = 'PDF Export failed!';
            messageEl.classList.remove('bg-yellow-500');
            messageEl.classList.add('bg-red-500');
        } finally {
            // Restore map controls
            controls.forEach(c => c.style.visibility = 'visible');
            
            // Hide message
            setTimeout(() => {
                messageEl.style.opacity = '0';
                messageEl.style.transform = 'translateX(100%)';
            }, 3000);
        }
    }
